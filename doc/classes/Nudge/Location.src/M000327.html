    def promote(popIndex, newLoc = nil)
      if newLoc
        raise(ArgumentError,
          "\"#{@name}\" is not connected to \"#{newLoc}\"") if !@downstream.include?(newLoc)
      else
        newLoc = @downstream.to_a.sample || self.name
      end
      self.transfer(popIndex,newLoc)
    end
