  def self.random_value(params = {})
    bottom = params[:randomFloatLowerBound] || @defaultLowest
    top = params[:randomFloatUpperBound] || @defaultHighest
    bottom, top = [bottom,top].min, [bottom,top].max
    range = top - bottom
    (rand*range) + bottom
  end
